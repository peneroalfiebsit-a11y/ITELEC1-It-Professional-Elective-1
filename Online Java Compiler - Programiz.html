import java.util.Scanner; public class group14 { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); // Prompt user for input System.out.print("Enter first number: "); double num1 = scanner.nextDouble(); System.out.print("Enter second number: "); double num2 = scanner.nextDouble(); System.out.print("Enter an operator (+, -, *, /): "); char operator = scanner.next().charAt(0); double result; // Perform arithmetic operations based on the operator if (operator == '+') { result = num1 + num2; System.out.println("Result: " + num1 + " + " + num2 + " = " + result); } else if (operator == '-') { result = num1 - num2; System.out.println("Result: " + num1 + " - " + num2 + " = " + result); } else if (operator == '*') { result = num1 * num2; System.out.println("Result: " + num1 + " * " + num2 + " = " + result); } else if (operator == '/') { if (num2 != 0) { result = num1 / num2; System.out.println("Result: " + num1 + " / " + num2 + " = " + result); } else { System.out.println("Error: Division by zero is not allowed."); } } else { System.out.println("Error: Invalid operator. Please use +, -, *, or /."); } // Close the scanner scanner.close(); } }
